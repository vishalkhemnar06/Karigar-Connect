# Razorpay Payment Gateway Configuration
# Add these to your .env file in the server directory

# Razorpay API Keys (get from https://dashboard.razorpay.com)
RAZORPAY_KEY_ID=your_razorpay_key_id_here
RAZORPAY_KEY_SECRET=your_razorpay_key_secret_here

# Optional: Webhook secret for handling payment callbacks
RAZORPAY_WEBHOOK_SECRET=your_webhook_secret_here

# Payment Settings
PAYMENT_MODE=production # or 'test' for testing

# Allowed Payment Methods
RAZORPAY_PAYMENT_METHODS=card,upi,netbanking,wallet

# Refund Policy (in days)
REFUND_ALLOWED_DAYS=7
